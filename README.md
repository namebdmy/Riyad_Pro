## в—Ҳ Command Structure

Every command lives in `scripts/cmds/` as a `.js` file. Here's the full structure:

```js
module.exports = {
  config: {
    name: "commandname",        // Command name (used as trigger)
    version: "1.0.0",           // Version of the command
    author: "YourName",         // Author name
    countDown: 5,               // Cooldown in seconds
    role: 0,                    // Permission level (see below)
    shortDescription: "...",    // Short description
    longDescription: "...",     // Long description
    category: "fun",            // Category shown in help
    guide: "{prefix}commandname [args]"  // Usage guide
  },

  // в”Җв”Җв”Җ Runs when command is triggered в”Җв”Җв”Җ
  onStart: async function ({ api, event, args, message, getLang }) {
    // event.threadID  вҶ’ thread/group ID
    // event.senderID  вҶ’ sender's Facebook ID
    // event.body      вҶ’ full message text
    // args            вҶ’ array of arguments after command name
    // message.reply() вҶ’ reply to the message
    // api.sendMessage() вҶ’ send a message anywhere

    message.reply("Hello!");
  },

  // в”Җв”Җв”Җ Runs when someone replies to the bot's message в”Җв”Җв”Җ
  onReply: async function ({ api, event, Reply, message }) {
    // Reply.author    вҶ’ original command caller's ID
    // Reply.messageID вҶ’ ID of the bot's message being replied to
    message.reply(`You replied: ${event.body}`);
  },

  // в”Җв”Җв”Җ Runs when someone reacts to the bot's message в”Җв”Җв”Җ
  onReaction: async function ({ api, event, Reaction, message }) {
    // event.reaction  вҶ’ the emoji used
    message.reply(`You reacted with: ${event.reaction}`);
  },

  // в”Җв”Җв”Җ Runs on every message in a thread (no prefix needed) в”Җв”Җв”Җ
  onChat: async function ({ api, event, message }) {
    if (event.body === "hello") message.reply("hi!");
  },

  // в”Җв”Җв”Җ Runs on thread events (join, leave, etc.) в”Җв”Җв”Җ
  onEvent: async function ({ api, event, message }) {
    if (event.logMessageType === "log:subscribe") {
      message.reply("Welcome!");
    }
  }
};
```

## в—Ҳ onReply Pattern

```js
// In onStart вҖ” send a message and register a reply listener
const sent = await message.reply("What's your name?");
global.GoatBot.onReply.set(sent.messageID, {
  commandName: "mycommand",
  messageID: sent.messageID,
  author: event.senderID,
  // any extra data you want to pass
  step: 1
});

// In onReply вҖ” handle the reply
onReply: async function ({ api, event, Reply, message }) {
  if (event.senderID !== Reply.author) return; // only original caller
  message.reply(`Hello, ${event.body}!`);
}
```
